234. Palindrome Linked List
Input: head = [1,2,2,1]
Output: true

class Solution {
    public boolean isPalindrome(ListNode head) {
        ListNode l=head;
        ArrayList<Integer> h=new ArrayList<>();
        while(head!=null){
            h.add(head.val);
            head=head.next;
        } 
         int left=0;
         int right=h.size()-1;
         while(left<right){
            if(h.get(left)!=h.get(right)){
                return false;
                    }
            left++;
            right--;
         }
         return true;

    }
}
