151. Reverse Words in a String

Input: s = "the sky is blue"
Output: "blue is sky the"

class Solution {
    public String reverseWords(String s) {
        String [] b=s.trim().split("\\s+");
        String ans="";
        for(int i=b.length-1;i>=0;i--){
            ans+=b[i]+" ";
        }
        return ans.trim();
    }
}
