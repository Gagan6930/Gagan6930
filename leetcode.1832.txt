
1832. Check if the Sentence Is Pangram

class Solution {
    public boolean checkIfPangram(String sentence) {
        HashSet<Character> h=new HashSet<>();
        for(char ch :sentence.toCharArray()){
            h.add(ch);
        }
        if(h.size()==26){
            return true;
        }else{
            return false;
        }
    }
}
