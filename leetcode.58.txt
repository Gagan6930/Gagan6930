class Solution {
    public int lengthOfLastWord(String s) {
        int i=s.length()-1;
        while(i>=0 && s.charAt(i)== ' '){.///trims the spaces of the string in the last 
            i--;
        }
        int count=0;
        while(i>=0 && s.charAt(i)!= ' '){///counts the charcters of the last word in the  String utill the spaces found in the String
            count++;
            i--;
        }
        return count;
    }
}
