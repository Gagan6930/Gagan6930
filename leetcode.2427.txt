2427. Number of Common Factors

class Solution {
    public int commonFactors(int a, int b) {
        int count=0;
        for(int i=1;i<=a;i++){   //check until the one of the value is ok...............
            if(a%i==0 && b%i==0){  //if i is factor of the both the a and b..............
count++;
            }
        }
        return count;
    }
}
