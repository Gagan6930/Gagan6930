class Solution {
    public int pivotIndex(int[] nums) {
        int t_s=0;
        for(int i:nums){
            t_s=t_s+i;//TOTAL SUM OF THE ARRAY ELEMENTS
        }
        int ls=0;
        for(int i=0;i<nums.length;i++){
            if(ls==t_s-ls-nums[i]){//CHECK WEATHER THE ELEMENTS LEFT SIDE ELEMNTS ARE EQUALS ARE NOT 
                return i;
            }else{
                ls+=nums[i];//ADDING THE NOT EQUALS NUMBER TO LEFT SUM
            }
           

        }
         return -1;
    }
}
