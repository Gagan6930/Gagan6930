204. Count Primes

Input: n = 10
Output: 4
Explanation: There are 4 prime numbers less than 10, they are 2, 3, 5, 7.


class Solution {
    public int countPrimes(int n) {
        int c=0;
        int a[]=new int[n];
        for(int i=2;i<=n;i++){
            for(int j=i+i;j<n;j=j+i){
                  a[j]=1;
            }
        }
        for(int k=2;k<n;k++){
            if(a[k]==0){
                c++;
            }
        }
        return c;
    }
}
