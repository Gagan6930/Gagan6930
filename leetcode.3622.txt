class Solution {
    public boolean checkDivisibility(int n) {
        int x=n;
        int sum=0;
        int pro=1;
        
        while(n>0){
         int re=n%10;
            sum+=re;
            pro=re;
            n=n/10;
        }
        
       
            return x%(sum+pro)==0;
             }
    
}
