344. Reverse String
//need to reverse the String 

class Solution {
    public void reverseString(char[] s) {
        int i=0;
        int r=s.length-1;
        while(i<r){    //check util the they will get the intersection
            char c=s[i];   //swap like temp
            s[i]=s[r];
            s[r]=c;
            i++;       //keep going util the while loop get false...
            r--;
        }
    }
}
