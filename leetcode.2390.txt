2390. Removing Stars From a String

Input: s = "leet**cod*e"
Output: "lecoe"

class Solution {
    public String removeStars(String s) {
       StringBuilder sb=new StringBuilder();
        for(char c:s.toCharArray()){
            if(c=='*'){
                sb.deleteCharAt(sb.length()-1);
            }else{
                sb.append(c);
            }
        } 
        return sb.toString();
           }
}
