203. Remove Linked List Elements

class Solution {
    public ListNode removeElements(ListNode head, int val) {
        ListNode l=head;
       ArrayList<Integer> k=new ArrayList<>();  //load all the values into the arraylist expect that target val.................
       while(head!=null){
        if(head.val!=val){
            k.add(head.val);

        }
        head=head.next;
       }
      ListNode dami=null;  //create one dami node for store the values of the arrayList which we have stored...
       for(int i:k){
        ListNode temp =new ListNode(i); //creating the one temp Listnode for the every value present in the arrayList..........
        if(dami==null){
            dami=temp;
            head=temp;  //also point the head to stsrting point of the dami node............
        }else{
            dami.next=temp;  //assigning the next value address to the prev node ......
            dami=dami.next;  //move to the store the next node from the arrayList
        }
       }
       return head;
    }
}
